expr -> expr ADD termo
      | expr MINUS termo
      | termo

termo -> termo MULT factor
       | termo DIV factor
       | termo MOD factor
       | factor

factor -> ID
        | REAL
        | INTEGER
        | STRING
        | OPENPARENTHESIS expr CLOSEPARENTHESIS
        | id OPENPARENTHESIS arguments CLOSEPARENTHESIS
        | array_getter
        | condition
